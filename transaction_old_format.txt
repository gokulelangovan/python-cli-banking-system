@app.get("/my-transactions/{account_id}")
def my_transactions(
    account_id: int,
    service=Depends(get_banking_service),
    user_id: int = Depends(get_current_user)
):
    customer_id = get_customer_id_by_user(user_id)

    account = account_repo.get_account_by_id(account_id)

    if not account:
        raise HTTPException(status_code=404, detail="Account not found")

    if account["customer_id"] != customer_id:
        raise HTTPException(status_code=403, detail="Unauthorized")

    transactions = service.get_account_transactions(account_id)

    return {"transactions": [dict(t) for t in transactions]}
